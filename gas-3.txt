int smokesensor=0;
void setup()
{
  pinMode(A0, INPUT);
  Serial.begin(9600);
  pinMode(5, OUTPUT);
  pinMode(7, OUTPUT);
}

void loop()
{
  smokesensor=analogRead(A0);
  Serial.println(smokesensor);
  if(smokesensor>=100){
    digitalWrite(5, HIGH);
  }
  else {
  // Wait for 1000 millisecond(s)
  digitalWrite(5, LOW);
  }
  if(smokesensor>=120){
    digitalWrite(7,HIGH);
  }
    else {
      digitalWrite(7,LOW);
    }
  delay(10);
  }
